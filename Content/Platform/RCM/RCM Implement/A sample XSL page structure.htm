A Sample XSL Page Structure

The following is a sample XSL.

<xsl:stylesheet version="1.0"

    xmlns:xsl="http://www.w3.org/1999/XSL/Transform"

    xmlns:actimize="com.actimize.infrastructure.util.XsltExtensionsFunctions">

 

<!-- WIDGET PARAMS -->

<xsl:variable name="optionalParams">

...

</xsl:variable>

 

<!-- HTML DEFINITION -->

<xsl:template match="/">

<head>

<script language="javascript">

function drawWidget() {

RCMAPI.WidgetsManager.getWidgetById("w1").draw();

}

window.attachEvent('onload', drawWidget);

</script>

<xsl:value-of

select="actimize:createWidget('w1', 'widgetdiv1', 'pivot_accounts', $optionalParams)" disable-output-escaping="yes" />

</head>

<body>

<div id="widgetdiv1" style="width: 300px; height: 300px" />

</body>

</xsl:template>

</xsl:stylesheet>
