// Step 1 - Create a cookie container
System.Net.CookieContainer cookieContainer = new System.Net.CookieContainer();

#region Step 2 - Log in
  AuthenticationWebService.ACMResult result;

  ACMAuthenticationServiceService authenticationWebService = new
ACMAuthenticationServiceService();
  // Assign the cookie container
  authenticationWebService.CookieContainer = cookieContainer;
  // Log in
  result = authenticationWebService.login( "admin", "password" );
  if (!result.Step) Console.WriteLine( "Error: Logging failed" );

#endregion
